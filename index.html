    var graph;

    $(document).ready(function(){
        graph = new Grapheon({
            id:'canvas',
            canvas:{
                width:500,
                height:500
            },
            processing_algorithm: processingAlgorithm1
        });

        graph.addNode("id1","Moscow")
                .addNode("id2","Paris")
                .addNode("id3","London");

        graph.addConnection("id1","id2")
                .addConnection("id1","id3")
                .addConnection("id3","id2");

        graph.draw().process();

    });
</script>
